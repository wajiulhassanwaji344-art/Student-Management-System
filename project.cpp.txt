#include <iostream>
using namespace std;

// -------- Function Declaration --------
void inputStudents(int id[], string name[], int math[], int phy[], int chem[]);
void displayStudents(int id[], string name[], int math[], int phy[], int chem[]);
void searchStudent(int id[], string name[], int math[], int phy[], int chem[]);

// -------- Main Function --------
int main() {

    int id[3];
    string name[3];
    int math[3], phy[3], chem[3];

    inputStudents(id, name, math, phy, chem);
    displayStudents(id, name, math, phy, chem);
    searchStudent(id, name, math, phy, chem);

    return 0;
}

// -------- Input Function --------
void inputStudents(int id[], string name[], int math[], int phy[], int chem[]) {

    for(int i = 0; i < 3; i++) {
        cout << "\nEnter data for student " << i + 1 << endl;

        cout << "ID: ";
        cin >> id[i];

        cout << "Name: ";
        cin >> name[i];

        cout << "Math marks: ";
        cin >> math[i];

        cout << "Physics marks: ";
        cin >> phy[i];

        cout << "Chemistry marks: ";
        cin >> chem[i];
    }
}

// -------- Display Function --------
void displayStudents(int id[], string name[], int math[], int phy[], int chem[]) {

    cout << "\n----- Student Record -----\n";

    for(int i = 0; i < 3; i++) {

        int total = math[i] + phy[i] + chem[i];
        float percentage = (total / 300.0) * 100;

        cout << "\nID: " << id[i];
        cout << "\nName: " << name[i];
        cout << "\nMath: " << math[i];
        cout << "\nPhysics: " << phy[i];
        cout << "\nChemistry: " << chem[i];
        cout << "\nTotal: " << total;
        cout << "\nPercentage: " << percentage << "%\n";
    }
}

// -------- Search Function --------
void searchStudent(int id[], string name[], int math[], int phy[], int chem[]) {

    int searchID;
    cout << "\nEnter ID to search: ";
    cin >> searchID;

    for(int i = 0; i < 3; i++) {

        if(id[i] == searchID) {

            int total = math[i] + phy[i] + chem[i];
            float percentage = (total / 300.0) * 100;

            cout << "\n--- Student Found ---";
            cout << "\nName: " << name[i];
            cout << "\nMath: " << math[i];
            cout << "\nPhysics: " << phy[i];
            cout << "\nChemistry: " << chem[i];
            cout << "\nPercentage: " << percentage << "%";
            return;
        }
    }

    cout << "Student not found!";
}
